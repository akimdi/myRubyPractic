/**
 *  ВАЛЮТНЫЙ КАЛЬКУЛЯТОР ( SME )
 *
 *  @update   2016-02-08
 */
$(function(){

    $.fn.rbSmeRatesCalculator = function(options){

        // приём настроек
        var settings = $.extend({'debug':false},options);

        // селекторы и значения
        var debug = settings.debug,
            fromValueSel = settings.fromValue,
            fromCurrencySel = settings.fromCurrency,
            from_value = fromValueSel.val(),
            from_currency = fromCurrencySel.val(),
            toValueSel = settings.toValue,
            toCurrencySel = settings.toCurrency,
            to_value = toValueSel.val(),
            to_currency = toCurrencySel.val(),
            max_length_input = fromValueSel.attr('maxlength') ? parseInt( fromValueSel.attr('maxlength') ) : 10,
            max_tabs_input = Math.floor(max_length_input/4);

        var rate = 0;

        // преобразование числа в читаемый вид ( узкий пробел &thinbsp; => String.fromCharCode(8201) )
        var toNiceNum = function(num){
            return num.toString().replace(/(\d+)(\.\d+)?/g,function(c,b,a){return b.replace(/(\d)(?=(\d{3})+$)/g,'$1'+String.fromCharCode(8201))+(a?a:'')});
        };

        // печать отладочной информации
        var debugOutput = function(from,to,sum,units,result){
            if ( debug ) {
                console.log(from+' => '+to+', sum: '+sum+', units: '+units+', rate: '+rate+', result: '+result);
            }
        };

        // получить ставку и единицу
        var getRateUnits = function(type,primary,secondary){
            return [Number($('span[data-mark="'+type+'-'+primary+'-'+secondary+'"]').text()),Number($('span[data-mark="'+type+'-'+primary+'-'+secondary+'"]').data('units'))];
        };

        // старт расчёта
        var runCalculate = function(){

            // одинаковая валюта
            if ( from_currency == to_currency ) {
                to_value = from_value;
            }
            else {
                // из одной в другую
                switch ( from_currency ) {

                    // из рублей
                    case 'RUR':
                        rate = getRateUnits('S',to_currency,from_currency)[0];
                        to_value = from_value / rate;
                        break;
                        break;

                    default:
                        rate = getRateUnits('B',from_currency,to_currency)[0];
                        to_value = from_value * rate;
                        break;
                }

                debugOutput(from_currency,to_currency,from_value,units,to_value);
            }

            // вставка результата
            toValueSel.val( toNiceNum( Number( to_value ).toFixed(2) ) );
        };

        // событие : изменение суммы конвертации
        fromValueSel.on('keyup change blur',function(event){
            // фильтруем ввод
            var str = $(this).val();
            str = str.replace(',','.');
            if ( str.indexOf('.')+1 > 0 ) {
                var str_arr = str.split('.',2);
                str_arr[0] = str_arr[0].replace(/\D+/g,'');
                if ( str_arr[0] == '' ) str_arr[0] = '0';
                str = str_arr[0].replace(/\D+/g,'')+'.'+str_arr[1].replace(/\D+/g,'').substr(0,2);
            } else {
                str = str.replace(/\D+/g,'');
            }
            // отдельно для change
            if ( event.type != 'keyup' ) {
                var regexp = /\.$/g;
                str = str.replace(regexp,'');
            }
            // обрезаем длину
            if ( str.length > max_length_input-max_tabs_input ) {
                str = str.substr(0,max_length_input-max_tabs_input);
            }
            // подставляем в input
            $(this).val( toNiceNum( str ) );
            // берем число для расчётов
            from_value = Math.abs( parseFloat( Number(str) ) );
            runCalculate();
        });

        // событие : изменение исходной валюты
        fromCurrencySel.on('change',function(){
            from_currency = $(this).val();
            runCalculate();
        });

        // событие : изменение желаемой валюты
        toCurrencySel.on('change',function(){
            to_currency = $(this).val();
            runCalculate();
        });

        // имитируем событие, для начального расчёта
        fromValueSel.trigger('change');
    };

    // вызов
    $('#smecalc_from_value').rbSmeRatesCalculator({
        // отладка
        'debug' : false,
        // подвязываем селекторы
        'fromValue'    : $('#smecalc_from_value'),    // поле input
        'fromCurrency' : $('#smecalc_from_currency'), // поле select
        'toValue'      : $('#smecalc_to_value'),      // поле input
        'toCurrency'   : $('#smecalc_to_currency'),   // поле select
    });

});
</script>

        $this.find('input[type="text"]').width(settings.width-175);
        $this.find('button').prop('disabled',true);

            if ( $this.css('display') == 'none' ){
                $this.width(0).height(settings.height).css('display','block').animate({'width':settings.width},settings.speed,function(){
                    $this_close.css('display','block');
                    $this.find('input').focus();
                });

                    $this.height(0).css('display','none');
                });
            }
            return false;
        });

            $this.animate({'width':0},settings.speed,function(){

            $this.find('input').width(settings.width-175);
        });
    };

});

<script type="text/javascript">
$(function(){
    $.fn.rbHeaderMap = function(options){
        var settings = $.extend({'bind':'li.all a','tabset':'div.switcher','content':'div.switcher,div.switcher_view,div.sitemap-wrap','speed':400},options);
        var $this = $(this);
        var $this_bind = $(settings.bind);
        var $this_close = $($this.selector+'_close');
        var $this_tabset = $this.find(settings.tabset);
        var $this_content = $this.find(settings.content);
        var $this_wrapper = $('#wrapper');
        var $this_sidebar = $('#sidebar');
        var setTabByCookie = function(){
            if ( $.cookie ) {
                switch ( $.cookie('tab_prefer') ) {
                    case 'for_life':
                        $('.switcher_item.for_life_view .pseudo').trigger('click');
                        break;
                    case 'for_business':
                        $('.switcher_item.for_business_view .pseudo').trigger('click');
                        break;
                }
            }
        }
        if ( $this_tabset.length ) {
            $this_tabset.off('resize').css({'width':0,'height':0,'margin-bottom':'50px'});
        }
        var max_width = parseInt($this.css('max-width'));
        $this_bind.click(function(){
            if ( $this.hasClass('not_display') ) {
                $this.css({'display':'none'}).removeClass('not_display');
            }            
            setTabByCookie();
            $this.css({'overflow':'auto','top':window.pageYOffset});
            settings.width = $this_wrapper.width()-$this_sidebar.width()+8;
            if ( settings.width > max_width ) {
                settings.width = max_width;
            }
            if ( $this_bind.hasClass('selected') ) return false;
            if ( $this.css('display') == 'none' ){
                $('.b-popup').hide();
                $this.css('height',$this.height());
                $this_content.css('opacity',0);
                $this_wrapper.css('opacity',0.25);
                $this.width(0).css('display','block');
                $this.animate({'width':settings.width},settings.speed,function(){
                    $this.css('height','auto');
                    $this_content.css('opacity',1);
                    $this_close.show();
                });
                $(window).bind('keypress.rbheadermap',function(e){
                    if ( e.keyCode == 27 ){
                        $this_close.trigger('click');
                    }
                });
            } else {
                $this_close.trigger('click');
            }
            return false;
        });
        $this_close.click(function(){
            $(window).unbind('keypress.rbheadermap');
            setTabByCookie();
            $this_close.hide();
            $this.css({'overflow':'hidden','height':$this.height()});
            $this_content.css('opacity',0);
            $this.animate({'width':0},settings.speed,function(){
                $this.hide().css('width','auto');
                $this_wrapper.animate({'opacity':1},parseInt(settings.speed)/2,function(){});
            });
            return false;
        });
    };
    $('#header-map').rbHeaderMap({'bind':'li.all a'});
    $('#header-request').rbHeaderMap({'bind':'li.online a'});
});
